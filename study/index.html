1. Singly Linked List

Architecture: Begins with a head pointer. Each node contains data and a single next pointer holding the memory address of the adjacent node. Strictly terminates with a nullptr.

💡 Mnemonic: "O.W.S." (One Way Street)
You can only drive forward. If you miss a house, you have to go all the way back to the start. 
class SinglyLinkedList {
private:
    struct Node { int data; Node* next; Node(int v) : data(v), next(nullptr) {} };
    Node* head;
public:
    SinglyLinkedList() : head(nullptr) {}

    void insert_start(int val) {
        Node* newNode = new Node(val);
        newNode->next = head;
        head = newNode;
    }
};
class SinglyLinkedList {
private:
    struct Node { int data; Node* next; Node(int v) : data(v), next(nullptr) {} };
    Node* head;
public:
    SinglyLinkedList() : head(nullptr) {}

    void insert_end(int val) {
        Node* newNode = new Node(val);
        if (!head) { head = newNode; return; }
        
        Node* temp = head;
        while (temp->next) temp = temp->next;
        temp->next = newNode;
    }
};
class SinglyLinkedList {
private:
    struct Node { int data; Node* next; Node(int v) : data(v), next(nullptr) {} };
    Node* head;
public:
    SinglyLinkedList() : head(nullptr) {}

    void insert_after(int pos, int val) {
        Node* temp = head;
        for(int i = 0; i < pos && temp; i++) temp = temp->next;
        
        if (temp) {
            Node* newNode = new Node(val);
            newNode->next = temp->next;
            temp->next = newNode;
        }
    }
};
class SinglyLinkedList {
private:
    struct Node { int data; Node* next; Node(int v) : data(v), next(nullptr) {} };
    Node* head;
public:
    SinglyLinkedList() : head(nullptr) {}

    void delete_val(int val) {
        if (!head) return;
        if (head->data == val) {
            Node* toDelete = head;
            head = head->next;
            delete toDelete;
            return;
        }
        Node* temp = head;
        while (temp->next && temp->next->data != val) temp = temp->next;
        
        if (temp->next) {
            Node* toDelete = temp->next;
            temp->next = temp->next->next;
            delete toDelete;
        }
    }
};
class SinglyLinkedList {
private:
    struct Node { int data; Node* next; Node(int v) : data(v), next(nullptr) {} };
    Node* head;
public:
    SinglyLinkedList() : head(nullptr) {}

    void reverse() {
        Node* prev = nullptr;
        Node* current = head;
        Node* next = nullptr;
        
        while (current != nullptr) {
            next = current->next;
            current->next = prev; 
            prev = current;       
            current = next;       
        }
        head = prev; 
    }
};

2. Circular Singly Linked List

Architecture: Like a singly list, but the last node points back to the first node's memory address instead of nullptr.

💡 Mnemonic: "R.R.R." (Round Robin Roulette)
Once you pass the last spot, you land right back on the first node. There is no end. 
class CircularSinglyList {
private:
    struct Node { int data; Node* next; Node(int v) : data(v), next(nullptr) {} };
    Node* tail; // We track the tail! Because tail->next is the head.
public:
    CircularSinglyList() : tail(nullptr) {}

    void insert_start(int val) {
        Node* newNode = new Node(val);
        if (!tail) {
            tail = newNode;
            tail->next = tail;
        } else {
            newNode->next = tail->next;
            tail->next = newNode;
        }
    }
};
class CircularSinglyList {
private:
    struct Node { int data; Node* next; Node(int v) : data(v), next(nullptr) {} };
    Node* tail;
public:
    CircularSinglyList() : tail(nullptr) {}

    void insert_end(int val) {
        Node* newNode = new Node(val);
        if (!tail) {
            tail = newNode;
            tail->next = tail;
        } else {
            newNode->next = tail->next;
            tail->next = newNode;
            tail = newNode; // Only difference! Update tail
        }
    }
};
class CircularSinglyList {
private:
    struct Node { int data; Node* next; Node(int v) : data(v), next(nullptr) {} };
    Node* tail;
public:
    CircularSinglyList() : tail(nullptr) {}

    void delete_start() {
        if (!tail) return;
        Node* head = tail->next;
        
        if (tail == head) { 
            delete head;
            tail = nullptr;
        } else {
            tail->next = head->next;
            delete head;
        }
    }
};

3. Doubly Linked List

Architecture: Two pointers per node! Forward and backward traversal via next and prev memory addresses.

💡 Mnemonic: "T.W.A." (Two Way Avenue)
You can walk up the avenue, drop your keys, and instantly walk backwards without going around the block. 
class DoublyLinkedList {
private:
    struct Node { int data; Node* next; Node* prev; Node(int v) : data(v), next(nullptr), prev(nullptr) {} };
    Node* head; Node* tail;
public:
    DoublyLinkedList() : head(nullptr), tail(nullptr) {}

    void insert_start(int val) {
        Node* newNode = new Node(val);
        if (!head) { head = tail = newNode; return; }
        newNode->next = head;
        head->prev = newNode;
        head = newNode;
    }
};
class DoublyLinkedList {
private:
    struct Node { int data; Node* next; Node* prev; Node(int v) : data(v), next(nullptr), prev(nullptr) {} };
    Node* head; Node* tail;
public:
    DoublyLinkedList() : head(nullptr), tail(nullptr) {}

    void insert_end(int val) {
        Node* newNode = new Node(val);
        if (!tail) { head = tail = newNode; return; }
        tail->next = newNode;
        newNode->prev = tail;
        tail = newNode;
    }
};
class DoublyLinkedList {
private:
    struct Node { int data; Node* next; Node* prev; Node(int v) : data(v), next(nullptr), prev(nullptr) {} };
    Node* head; Node* tail;
public:
    DoublyLinkedList() : head(nullptr), tail(nullptr) {}

    void insert_after(Node* target, int val) {
        if (!target) return;
        Node* newNode = new Node(val);
        
        newNode->next = target->next;
        newNode->prev = target;
        
        if (target->next) target->next->prev = newNode;
        else tail = newNode; 
        
        target->next = newNode;
    }
};
class DoublyLinkedList {
private:
    struct Node { int data; Node* next; Node* prev; Node(int v) : data(v), next(nullptr), prev(nullptr) {} };
    Node* head; Node* tail;
public:
    DoublyLinkedList() : head(nullptr), tail(nullptr) {}

    void delete_node(Node* node) {
        if (!node) return;
        
        if (node->prev) node->prev->next = node->next;
        else head = node->next;
        
        if (node->next) node->next->prev = node->prev;
        else tail = node->prev;
        
        delete node;
    }
};

4. Circular Doubly Linked List

Architecture: The ultimate structure. Doubly linked, but the last node connects to the first. No nullptr edge cases ever exist in a populated list.

💡 Mnemonic: "M.G.R." (Merry Go Round)
You can walk left (prev) or right (next) continuously without ever hitting an edge or a wall. 
class CircularDoublyList {
private:
    struct Node { int data; Node* next; Node* prev; Node(int v) : data(v), next(this), prev(this) {} };
    Node* head;
public:
    CircularDoublyList() : head(nullptr) {}

    void insert_start(int val) {
        Node* newNode = new Node(val);
        if (!head) { head = newNode; return; }
        
        Node* tail = head->prev;
        
        newNode->next = head;
        newNode->prev = tail;
        tail->next = newNode;
        head->prev = newNode;
        
        head = newNode; // Move head backward
    }
};
class CircularDoublyList {
private:
    struct Node { int data; Node* next; Node* prev; Node(int v) : data(v), next(this), prev(this) {} };
    Node* head;
public:
    CircularDoublyList() : head(nullptr) {}

    void insert_end(int val) {
        Node* newNode = new Node(val);
        if (!head) { head = newNode; return; }
        
        Node* tail = head->prev;
        
        newNode->next = head;
        newNode->prev = tail;
        tail->next = newNode;
        head->prev = newNode;
        
        // DO NOT move head backward. New node becomes tail automatically.
    }
};
class CircularDoublyList {
private:
    struct Node { int data; Node* next; Node* prev; Node(int v) : data(v), next(this), prev(this) {} };
    Node* head;
public:
    CircularDoublyList() : head(nullptr) {}

    void delete_node(Node* node) {
        if (!node) return;
        
        if (node->next == node) { 
            head = nullptr;
        } else {
            node->prev->next = node->next;
            node->next->prev = node->prev;
            if (head == node) head = node->next;
        }
        delete node;
    }
};
